<script>
// --- VARIABLES GLOBALES ---
let balance = 1000.00;
let totalWagered = 0;
let betHistory = [];
let selectedBet = null;

// --- INITIALISATION ---
window.onload = () => {
    loadPlayerData();
    loadMatches();
};

function startApp() { document.getElementById('welcomePage').style.display = 'none'; }

function showSec(id) {
    document.querySelectorAll('.section').forEach(s => s.classList.remove('active-sec'));
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active-link'));
    document.getElementById(id).classList.add('active-sec');
    if(id === 'profilePage') updateStats();
    document.getElementById('betSlip').style.display = 'none';
}

// --- CORE: UI & SAVE ---
function updateUI() {
    balance = Math.round(balance * 100) / 100;
    document.getElementById('mainBalance').innerHTML = `${balance.toFixed(2)} <span style="color:var(--primary); font-size:0.8rem;">TND</span>`;
    
    // VIP Progress
    const progress = (totalWagered % 5000) / 5000 * 100;
    document.getElementById('vipBar').style.width = progress + "%";
    const levelLabel = document.getElementById('vipLevel');
    if (totalWagered >= 10000) levelLabel.innerText = "💎 VIP GOLD";
    else if (totalWagered >= 2000) levelLabel.innerText = "🥈 VIP SILVER";
    
    savePlayerData();
}

function savePlayerData() {
    localStorage.setItem('tnBet_data', JSON.stringify({balance, totalWagered, betHistory}));
}

function loadPlayerData() {
    const saved = localStorage.getItem('tnBet_data');
    if(saved) {
        const data = JSON.parse(saved);
        balance = data.balance;
        totalWagered = data.totalWagered;
        betHistory = data.betHistory;
        updateUI();
    }
}

function addHistory(game, bet, status, payout) {
    betHistory.unshift({ date: new Date().toLocaleTimeString(), game, bet, status, payout });
    if(betHistory.length > 20) betHistory.pop();
    renderHistory();
}

function renderHistory() {
    const list = document.getElementById('historyList');
    if(!list) return;
    list.innerHTML = betHistory.map(h => `
        <div style="background:#0d141f; padding:10px; border-radius:8px; margin-bottom:5px; display:flex; justify-content:space-between; font-size:0.8rem; border-left:3px solid ${h.payout > 0 ? 'var(--primary)' : 'var(--danger)'}">
            <div><b>${h.game}</b><br><small style="color:#555">${h.date}</small></div>
            <div style="text-align:right"><b>${h.payout} TND</b><br><small>${h.payout > 0 ? 'GAGNÉ' : 'PERDU'}</small></div>
        </div>
    `).join('');
}

// --- JEU: MINES ---
let minesGame = { active: false, bombs: [], revealed: 0 };
function initMines() {
    if(minesGame.active) { cashOutMines(); return; }
    if(balance < 10) return alert("Solde insuffisant");
    
    balance -= 10; totalWagered += 10;
    minesGame = { active: true, revealed: 0, bombs: Array.from({length:25},(_,i)=>i).sort(()=>Math.random()-0.5).slice(0,3) };
    
    const grid = document.getElementById('minesGrid');
    grid.innerHTML = '';
    for(let i=0; i<25; i++) {
        const t = document.createElement('div');
        t.className = 'mine-tile';
        t.onclick = () => revealMine(i, t);
        grid.appendChild(t);
    }
    document.getElementById('minesBtn').innerText = "CASH OUT (0.00)";
    updateUI();
}

function revealMine(idx, el) {
    if(!minesGame.active || el.classList.contains('revealed')) return;
    el.classList.add('revealed');
    if(minesGame.bombs.includes(idx)) {
        el.innerHTML = '💣'; el.style.background = 'var(--danger)';
        minesGame.active = false;
        addHistory('MINES', 10, 'LOSS', 0);
        setTimeout(()=> { alert("BOOM!"); resetMinesUI(); }, 500);
    } else {
        el.innerHTML = '💎'; el.style.background = 'var(--primary)'; el.style.color = 'black';
        minesGame.revealed++;
        let gain = 10 * Math.pow(1.2, minesGame.revealed);
        document.getElementById('minesBtn').innerText = `CASH OUT (${gain.toFixed(2)})`;
    }
}

function cashOutMines() {
    let gain = 10 * Math.pow(1.2, minesGame.revealed);
    balance += gain; celebrate();
    addHistory('MINES', 10, 'WIN', gain);
    resetMinesUI(); updateUI();
}

function resetMinesUI() {
    minesGame.active = false;
    document.getElementById('minesBtn').innerText = "JOUER (10 TND)";
    document.getElementById('minesGrid').innerHTML = '';
}

// --- JEU: CRASH ---
let crashState = { running: false, multi: 1.0, limit: 0, timer: null };
function handleCrashBtn() {
    if(crashState.running) {
        let win = 10 * crashState.multi;
        balance += win; celebrate();
        addHistory('CRASH', 10, 'WIN', win);
        stopCrash();
    } else {
        if(balance < 10) return;
        balance -= 10; totalWagered += 10; updateUI();
        crashState = { running: true, multi: 1.0, limit: (Math.random() * 5 + 1).toFixed(2) };
        document.getElementById('crashBtn').innerText = "CASH OUT";
        crashState.timer = setInterval(() => {
            crashState.multi += 0.02;
            document.getElementById('crashMulti').innerText = crashState.multi.toFixed(2) + "x";
            if(crashState.multi >= crashState.limit) {
                document.getElementById('crashMulti').style.color = 'var(--danger)';
                addHistory('CRASH', 10, 'LOSS', 0);
                stopCrash();
            }
        }, 100);
    }
}
function stopCrash() {
    clearInterval(crashState.timer); crashState.running = false;
    document.getElementById('crashBtn').innerText = "PARIER 10 TND";
    setTimeout(()=> { document.getElementById('crashMulti').style.color = 'white'; }, 2000);
    updateUI();
}

// --- SPORT ---
const matches = [{t:"EST vs ESS", o:[1.8, 3.0, 4.0]}, {t:"CA vs CSS", o:[2.1, 2.9, 3.5]}];
function loadMatches() {
    document.getElementById('matchList').innerHTML = matches.map(m => `
        <div class="match-card">
            <span>${m.t}</span>
            <div>
                <button class="odd-btn" onclick="openSlip('${m.t}','1',${m.o[0]})">${m.o[0]}</button>
                <button class="odd-btn" onclick="openSlip('${m.t}','X',${m.o[1]})">${m.o[1]}</button>
                <button class="odd-btn" onclick="openSlip('${m.t}','2',${m.o[2]})">${m.o[2]}</button>
            </div>
        </div>
    `).join('');
}
function openSlip(t, s, o) {
    selectedBet = {t, s, o};
    document.getElementById('betSlip').style.display = 'block';
    document.getElementById('slipContent').innerHTML = `${t}<br>Prono: <b>${s}</b> | Cote: <b>${o}</b>`;
}
function placeSportsBet() {
    let amt = parseFloat(document.getElementById('betAmount').value);
    if(balance < amt) return;
    balance -= amt; totalWagered += amt;
    addHistory('SPORT: '+selectedBet.t, amt, 'WIN', amt*selectedBet.o); // Sim win pour démo
    document.getElementById('betSlip').style.display = 'none';
    celebrate(); updateUI();
}

// --- BONUS 12% ---
document.getElementById('depAmount').addEventListener('input', function() {
    let bonus = (this.value * 0.12).toFixed(2);
    document.getElementById('bonusPreview').innerText = `+ ${bonus} TND (Bonus 12%)`;
});
function processDeposit() {
    let amt = parseFloat(document.getElementById('depAmount').value);
    if(amt < 10) return alert("Min 10 TND");
    let total = amt * 1.12;
    balance += total; celebrate(); updateUI();
    alert("Compte crédité !");
}

// --- UTILS ---
function celebrate() {
    for(let i=0; i<30; i++) {
        const c = document.createElement('div');
        c.className = 'confetti';
        c.style.left = Math.random()*100+'vw';
        c.style.background = `hsl(${Math.random()*360}, 70%, 50%)`;
        c.style.animationDuration = (Math.random()*1+1)+'s';
        document.body.appendChild(c);
        setTimeout(()=>c.remove(), 2000);
    }
}
function updateStats() {
    document.getElementById('statWinRate').innerText = betHistory.length ? (betHistory.filter(h=>h.payout>0).length/betHistory.length*100).toFixed(0)+"%" : "0%";
    document.getElementById('statTotalWagered').innerText = totalWagered.toFixed(2);
    let profit = betHistory.reduce((a,b)=>a+(b.payout-b.bet),0);
    document.getElementById('statNetProfit').innerText = profit.toFixed(2);
}
function resetGame() { localStorage.clear(); location.reload(); }

// ROULETTE
function playRoulette(betColor) {
    if(balance < 10) return;
    balance -= 10; totalWagered += 10; updateUI();
    const deg = Math.floor(Math.random()*360) + 1440;
    document.getElementById('wheel').style.transform = `rotate(${deg}deg)`;
    setTimeout(()=> {
        let win = Math.random() > 0.7; // Sim simple
        if(win) { balance += 20; celebrate(); addHistory('ROULETTE', 10, 'WIN', 20); }
        else addHistory('ROULETTE', 10, 'LOSS', 0);
        updateUI();
    }, 4000);
}
